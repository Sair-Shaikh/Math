\begin{exercise}{3.2.D}
    If $k$ is a field, show that $\Spec k[x]$ has infinitely many points. 
\end{exercise}
\begin{solution}
    Assume there are only finitely many prime ideals in $k[x]$, $\mathfrak p_1, \cdots, \mathfrak p_n$. As $k[x]$ is a PID, let $p_1, \cdots, p_n$ be the generators of the ideals. Then, $p_1\cdots p_n + 1$ has gcd $1$ with every prime ideal, thus is not contained in any of them (else these ideals would be the whole ring). Thus, there is a maximal ideal that contains this that is not amongst our finite primes. Contradiction. 
\end{solution}

\begin{exercise}{3.2.G}
    Show that if $A$ is an integral domain that is a finite $k$-algebra, then $A$ is a field. 
\end{exercise}
\begin{solution}
    Define $\phi_a: A \to A$ as multiplication by $a$, where $a \neq 0$. Then, $\phi_a$ is a $k$-linear map. Moreover, it is injective as $A$ is an integral domain. Thus, it is an isomorphism. So there exists $y \in A$ such that $\phi_a(y) = ay = 1$.
\end{solution}


\begin{exercise}{3.2.J}
Suppose $A$ is a ring, and $I$ an ideal of $A$. Let $\phi \colon A \to A/I$.
Show that $\phi^{-1}$ gives an inclusion-preserving bijection between prime
ideals of $A/I$ and prime ideals of $A$ containing $I$. Thus we can picture
$\Spec A/I$ as a subset of $\Spec A$.
\end{exercise}
\begin{solution}
    Let $J$ be an ideal in $A/I$. We first show that $\phi^{-1}(J)$ is an ideal containing $I$. \bbni
    First, note that $0 \in J$, thus, $I \in \phi^{-1}(J)$. Next, take $f, g \in \phi^{-1}(J)$. Then, $\phi(f), \phi(g) \in J$, thus $\phi(f)+\phi(g) = \phi(f+g) \in J$. Thus, $f+g \in \phi^{-1}(J)$. Similarly, take $a \in A$ and $f \in \phi^{-1}(J)$. Then, $\phi(af) \in J$ thus $af \in \phi^{-1}(J)$. \bbni

\end{solution}

\begin{exercise}{3.2.M}
    If $\phi: B \to A$ is a map of rings and $\mathfrak p$ is a prime ideal of $A$, show that $\phi^{-1}(\mathfrak p)$ is a prime ideal of $B$. 
\end{exercise}
\begin{solution}
    Let $f, g \in \phi^{-1}(\mathfrak p)$ and $b \in B$. First, note that $f + g \in \phi^{-1}(\mathfrak p)$ as $\phi(f+g) = \phi(f)+\phi(g) \in \mathfrak p$. Similarly, $af \in \phi^{-1}(\mathfrak p)$ as $\phi(af) = \phi(a)\phi(f) \in \mathfrak p$. Thus, $\phi^{-1}(\mathfrak p)$ is an ideal. Then, assume $fg \in \phi^{-1}(\mathfrak p)$. Then $\phi(fg) = \phi(f)\phi(g) \in \mathfrak p$. Thus, either $\phi(f)$ or $\phi(g)$ in $\mathfrak p$. Thus, $f \in \phi^{-1}(\mathfrak p)$ or $g \in \phi^{-1}(\mathfrak p)$. 
\end{solution}

\begin{exercise}{3.4.D}
    Show that $\sqrt{I}$ is an ideal. Show that $V(\sqrt{I}) = V(I)$. 
\end{exercise}
\begin{solution}
    Consider the map $R \to R/I$. This maps $\sqrt{I} \to \mathfrak{R}$. By $(3.2.R(b))$, $\mathfrak R$ is an ideal. Thus, its preimage is an ideal containing $I$. However, the preimage is exactly $\sqrt{I}$ by definition of a radical. \\
    Moreover, as $I \subseteq \sqrt{I}$, $V(\sqrt{I}) \subseteq V(I)$. Next, let $\mathfrak p \in V(I)$ and $f \in \sqrt{I}$ with $f^n \in I$. Then, as $I \subseteq \mathfrak p$, $f^n \in \mathfrak p$. Thus, $f \in \mathfrak p$ by primality.   
\end{solution}

\subsection{Base for the topology:}
\begin{exercise}{3.5.A}
    Show that the distinguished open sets form a base for the topology of $\Spec R$. 
\end{exercise}
\begin{solution}
    Let $U \subseteq \Spec R$ be an open set. Then, $\Spec R \setminus U$ is (Zariski) closed, thus can be written as $V(I)$ for some ideal $I$. We claim that $U = \bigcup_{f \in I} D(f)$. \\
    Let $\mathfrak p \in U$. Then, there is $f \in I$ such that $f \not \in \mathfrak p$. That is, $\mathfrak p \in D(f)$. \\
    Conversely, let $\mathfrak p \in D(f)$ for $f \in I$. Then, $f \not \in \mathfrak p$. Thus, $\mathfrak p \not \in V(I)$. Thus, $\mathfrak p \in U$. \\
    We are done. 
\end{solution}

\begin{exercise}{3.5.B}
Suppose $f_i \in A$ as $i$ runs over some index set $J$.
Show that
\[ \bigcup_{i \in J} D(f_i) = \Spec A \]
if and only if $(\{f_i\}_{i \in J}) = A$, or equivalently and very usefully,
if there are $a_i$ $(i \in J)$, all but finitely many $0$, such that
\[ \sum_{i \in J} a_i f_i = 1. \]
\end{exercise}
\begin{solution}
    Assume $(\{f_i\}_{i \in J}) = A$. Let $\mathfrak p \in \Spec A$ be a prime ideal. Then, there exists $i \in J$ such that $f_i \not \in \mathfrak p$ as $\mathfrak p \subsetneq A$. Thus, $\mathfrak p \in D(f_i)$. Thus, $\Spec A = \bigcup_{i \in J} D(f_i)$. \\
    Assume $\bigcup_{i \in J} D(f_i) = \Spec A$. Let $I = (\{f_i\}_{i \in J})$. If $I$ was a proper ideal of $A$, then $I \subseteq \mathfrak m$, where $\mathfrak m$ is some maximal ideal. However, $\mathfrak m \in D(f_i)$ for some $i$, thus, $f_i \not \in \mathfrak m$. Thus, $\mathfrak m$ cannot exist and $I$ cannot be proper. Thus, $I = A$. \\
    The latter fact follows from the definition of an ideal. 
\end{solution}

\begin{exercise}{3.5.C}
Show that if $\Spec A$ is an infinite union of distinguished open sets
$\bigcup_{i \in J} D(f_i)$, then in fact it is a union of a finite number
of these, i.e., there is a finite subset $J' \subset J$ so that
\[ \Spec A = \bigcup_{j \in J'} D(f_j) \]
\end{exercise}
\begin{solution}
    Assume $\Spec A = \bigcup_{i \in J} D(f_i)$. By 3.5.B, we know that $(\{f_i\}_{i \in J}) = A$. Thus, 
    \[ 1 = \sum_{i \in J' \subset J} a_if_i\]
    where $a_i \in A$ and $J'$ is finite. Then, $(\{f_i\})_{i \in J'} = A$. By 3.5.B, we have $\Spec A = \bigcup_{i \in J'} D(f_i)$. 
\end{solution}

\begin{exercise}{3.5.D}
    Show that
        \[ D(f) \cap D(g) = D(fg) \]
\end{exercise}
\begin{solution}
    Let $\mathfrak p \in D(fg)$. Then, $fg \not \in \mathfrak p$. Thus, $f,g \not \in \mathfrak p$. Thus, $\mathfrak p \in D(f) \cap D(g)$. \\
    Let $\mathfrak p \in D(f) \cap D(g)$. Then, $f, g \not \in \mathfrak p$. As $\mathfrak p$ is a prime ideal, $fg \not \in \mathfrak p$ (contrapositive of the prime property). Thus, $\mathfrak p \in D(fg)$. We are done. 
\end{solution}

\begin{exercise}{3.5.E}
    Show that $D(f) \subseteq D(g)$ if and only if $f^n \in (g)$ for some $n \geq 1$ if and only if $g$ is invertible in $A_f$. 
\end{exercise}
\begin{solution}
    First, we show $(2) \iff (3)$. Assume $f^n \in (g)$. Then, there exists $a \in A$ such that $f^n = ag$. Thus, in $A_f$, $g$ has inverse $\frac{a}{f^n}$. Similarly, assume $g$ is invertible in $A_f$. Then, there exists $\frac{a}{f^m} \in A_f$ with $a \in A$ satisfying $g\left(\frac{a}{f^n}\right) = 1$. Then, $ag = f^n$. \\
    Next, we show $(2) \implies (1)$. Assume $f^n \in (g)$. Let $\mathfrak p \in V(g)$. Then, $f^n \in \mathfrak p$ and thus $f \in \mathfrak p$. This implies $V(g) \subseteq V(f)$, or, taking the complement $D(f) \subseteq D(g)$. \\
    Finally, we show $(1) \implies (3)$. Assume $D(f) \subseteq D(g)$. Then, $V(g) \subseteq V(f)$. Thus, $f$ vanishes on $V(g)$. By 3.4.J, $f \in \sqrt{(g)}$, i.e. $f^n \in (g)$ for some $n$. 
\end{solution}

\begin{exercise}{3.5.F}
    Show that $D(f) = \emptyset$ if and only if $f \in \mathfrak R$. 
\end{exercise}
\begin{solution}
    This follows from $\mathfrak R$ being the intersection of all prime ideals. $f \in \mathfrak R$ if and only if it is contained in every prime ideal if and only if $D(f) = \emptyset$.
\end{solution}

\subsection{The Structure Sheaf: }
\begin{exercise}{4.1.A}
    Show that the natural map $A_f \to \OO_{\Spec A}(D(f))$ is an isomorphism.
\end{exercise}
\begin{solution}
    Note that $\OO_{\Spec A}(D(f))$ is the localization of $A$ at all functions $g$ such that $D(f) \subseteq D(g)$. From $(3.5.E)$, we know such $g$ are already invertible in $A_f$. Certainly, $A_f \subseteq \OO_{\Spec A}(D(f))$ (via appropriate further localization). Thus, $A_f \simeq \OO_{\Spec A}(D(f))$.
\end{solution}
