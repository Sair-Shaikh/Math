\begin{exercise}{3.6.I}
    Show that the closed points of $\Spec A$ correspond to maximal ideals of $A$. 
\end{exercise}
\begin{solution}
    Let $\mathfrak{p} \in \Spec A$ be point. Then, if $\mathfrak p$ is contained in some maximal ideal $\mathfrak m$. Thus, $V(\mathfrak{p})$ contains $\mathfrak m$. Then, it is easy to see that $\{\mathfrak p\}$ closed forces $\mathfrak p = \mathfrak m$, i.e. $\mathfrak p$ is maximal.
\end{solution}

\begin{exercise}{5.1.E}
    Show that if $X$ is a quasicompact scheme, then every point has a closed point in its closure. That is, every non-empty closed subset of $X$ contains a closed point of $X$.
\end{exercise}
\begin{solution}
    Let $Z$ be some non-empty irreducible closed subset of $X$. Let $U_1, \cdots, U_n$ be a finite affine open cover of $X$. Then, we have $Z = \bigcup_{i = 1}^n (Z \cap U_i)$, all open in $Z$. As $Z$ is irreducible, we must have $Z \cap U_i$ dense in $Z$ for some $i$. Moreover, $Z \cap U_i$ is a non-empty closed subset of the affine scheme $U_i$, so it contains a closed point.  
\end{solution}
