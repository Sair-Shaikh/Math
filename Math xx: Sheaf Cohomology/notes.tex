\section{Sheaf Cohomology:}
Recall that the global sections functor, $\Gamma$ is left-exact. 

Our goal is to take $X$, a separated, quasi-compact $A$-scheme and $F$ a quasi-coherent sheaf of $\OO_X$ modules, to produce sheaf cohomology groups $H(X, F)$ satisfying: 
\begin{enumerate}
    \item $H^i(X, -): QCoh \to Mod_A$ are functors. 
    \item $H^0(X, -) = \Gamma(X, -)$  as functors.
    \item Given exact seqeunce $0 \to F \to G \to H \to 0$, then we want the long exact sequence: 
    \[ 0 \to H^0(X, F) \to H^0(X, G) \to H^0(X, H) \to H^1(X, F) \to  H^1(X, G) \cdots \] 
    \item Given $\pi: X \to Y$ and $F$ a qcoh on $X$, you will get $H^i(Y, \pi_*(F)) \to H^i(X, F)$ extending the usual pullback map on global sections. 
    \item The map in (4) is an isomoprhism if $\pi$ is affine. 
    \item If $X$ covered by $n$ affines, then $H^1(X, F) = 0$ for $i \geq n$. In particular, $\Gamma$ is exact for $X$-affine.
    \item $H^i(X, \colim F_j) = \colim H^1(X, F_j)$. In particular, $H^i(X, F_1 \oplus F_2) = H^1(X, F_1) \oplus H^1(X, F_2)$. 
    \item The cohomology of line bundles on projective space is ... look at the next theorem.
\end{enumerate}

For projective space, we want the following:
\begin{theorem}
    .
    \begin{enumerate}
        \item $H^0(\P_A^n, \OO(m))$ is free of rank $\binom{n+m}{m}$ if $m \geq 0$. 
        \item $H^n(\P^n_A, \OO(m))$ is free of rank $\binom{-m-1}{-n-m-1}$ if $m \leq -n-1$
        \item $H^i(\P^n_A, \OO(m)) = 0$ in all other cases. 
    \end{enumerate}
\end{theorem}

Observe now the following: 
\begin{enumerate}
    \item These are all finitely generated $A$-modules. This is true for any coherent sheaf on a proper $A$-scheme. 
    \item These vanish for $i > n = \dim(\P^n_A)$. 
    \item The top cohomology, $H^n(\P^n_A, \OO(m))$ is $1$ dimensional for $m = -n-1$ and induces a perfect pairing:
    \[ H^i(\P_k^n, \OO(m)) \times H^{n-i}(\P^n_k, \OO(-n-1-m)) \to H^n(\P^n_k, \OO(-n-1)) = k\]
    This is a special case of Serre duality. $\OO(-n-1)$ is a special case of a ``dualizing sheaf''. 
    \item Higher cohomology vanishes for $m > -n-1$. This is a special case of the Kodaira vanishing theorem.
    \item $\Sigma_i (-1)^i h^i(\P_k^n, \OO(m))$ is a polynomial in $m$. This is called the Hilbert polynomial of projective space. 
\end{enumerate}

\begin{theorem}
    Let $X$ be projective over $A$ noetherian and $F$ a coherent sheaf on $X$. Then, 
    \begin{enumerate}
        \item The $H^i(X, F)$ are finitely generated $A$-modules
        \item The $H^i(X, F(m))$ vanish for $i > 0$ and $m >> 0$. Note: $F(m) = F \otimes \OO(m)$. 
    \end{enumerate}
\end{theorem}

\begin{proof}
    Step 1: Reduce to $X = \P_A^n$. For $\pi : X \hookrightarrow \P_A^n$, we have:
    \[ H^i(X, F(m)) =  H^1(\P^n_A, \pi_*(F(m)))\]
    because closed embeddings are affine. So we might as well do this for projective space. 

    Step 2: Present $F$ as: 
    \[ 0 \to G \to \OO(m)^j \to F \to 0 \]
    giving the long exact sequence:
    \[ 
    \xymatrix{
    0 \ar[r] & H^0(G) \ar[r] & H^0(\OO(m)^j) \ar[r] & H^0(F) \ar[lld] \\
    & H^1(G) \ar[r] & H^1(\OO(m)^j) \ar[r] & H^1(F) \ar[lld] \\
    & H^2(G) \ar[r] & H^2(\OO(m)^j) \ar[r] & H^2(F) \ar[lld] \\
    & \vdots \ar[r] & \vdots \ar[r] & \vdots \ar[lld] \\
    & H^{n-1}(G) \ar[r] & H^{n-1}(\OO(m)^j) \ar[r] & H^{n-1}(F) \ar[lld] \\
    & H^n(G) \ar[r] & H^n(\OO(m)^j) \ar[r] & H^n(F) \ar[r] & 0
    } \]


    Step 3: Backwards induction. 
    \begin{enumerate}
        \item $H^n(\OO(m)^j)$ finitely generated implies $H^n(F)$ finitely generated for all $F$ implies $H^n(G)$ finitely generated implies $H^{n-1}(F)$ finitely generated for all $F$. 
        \item Twist by $\OO(N)$ for $N >> 0$. Then, 
        \[ H^n(\OO(m+N)^j)= 0 \implies H^n(F(N)) = 0 \]
        for all $F$ and $N$ sufficiently large. This implies, that $H^n(G(N)) = 0$ for $N >>_F 0$ which implies $H^{n-1}(F(N)) = 0$. 
    \end{enumerate}
    In particular, we showed that $\Gamma(X, F)$ is a finitely generated $A$-module.
\end{proof}

\begin{corollary}
    Let $Y$ is locally noetherian. Then $\pi: X \to Y$ is finite if and only if it is affine and projective. 
\end{corollary}

\section{Euler Characteristics:}
Let $X$ be a projective $k$-scheme and $F$ a coherent sheaf on $X$. 

\begin{definition}
    The Euler characteristic is 
    \[ \chi(X, F) := \sum_{i = 0}{\dim X} (-1)^i h^i(X, F)\]
\end{definition}

\begin{example}
    \[ \chi(\P^n_k, \OO(m)) = \frac{(m+1)\cdots(m+n)}{n!} = \binom{m+n}{m}\]
\end{example}

Computing Euler characteristics is much easier than computing the ranks of the cohomology groups. 

\subsection{Riemann-Roch:}
Let $C$ projective curve and 
\[ D = \sum_{p \in C^{reg}} a_p[p]\]
be a divisor supported at regular points of $C$. 

\begin{definition}
    We define the degree, 
    \[ \deg(D) = \sum a_p \deg p\]
    where $\deg p$ is the degree of extension of the residue field. 
\end{definition}

\begin{definition}
    Let $\mathcal{L}$ be a line cunvle over $C$ a curve. Then, 
    \[ \deg \mathcal L =  \chi(C, \mathcal L) - \chi(C, \OO_C) \]
\end{definition}


In complex analysis, these curves look topologically like a surface. We count the number of holes. That is, the genus $g$ is the number of holes and is a topological invariant. Turns out, 
\[ \chi(C, \OO_C) = 1-g \]

Further, we can consider holomorphic line bundles and map them to topological line bundles. 
\begin{align*}
    \{\text{holomorphic line bundles }\} \to \{\text{topological line bundles}\} = H^2_{sing}(X) = \Z  
\end{align*} 
This is the degree map. 

\begin{definition}
    Let $X$-scheme of dimension $n$. The arithmetic genus is,
    \[ p_a(X) = (-1)^n (\chi(X, \OO_X)-1) \]
\end{definition}

Finally, we have the Riemann-Roch theorem,
\begin{theorem}
    \[ h^0(C, \LL) -h^1(C, \LL) = \deg \LL + 1 - p_a(C)\]
\end{theorem}
There is no content to this theorem as we state it now. 

More generally, for $F$ a coherent sheaf on $C$, 

\begin{definition}
    \[ \deg(F) = \chi(C, F) - (\rk F)\chi(C, \OO_C)\]
    where $\rk F$ is at the generic point. 
\end{definition}

We explain Serre Duality with an example.

    Near $p$ and $q$, we expand $f$,
    \begin{align*}
        f \sim_p \frac{a_{-1}}{x-2} + a_0 + a_1(x-2) + \cdots \\
        f \sim_q \frac{b_{-1}}{x-2} + b_0 + b_1(x-2) + \cdots 
    \end{align*}
    Up to addition of a constant, $f$ is determined by $a_{-1}$ and $b_{-1}$. We call $a_{-1}$ and $b_{-1}$ the principal parts of the function. 

    Write $\omega_C$ to be a line bundle of the algebraic/holomorphic differential forms on $C$. 

    Later we will give precise definitions and show that $\omega_C \cong \OO_C$ in this case. 

    Thus, in particular it has a unique section $\alpha$ up to scaling. 

    Claim: $\alpha = \frac{dx}{y} = \frac{2dy}{3x^2}$. This is regular and nonvanishing away from $L$ (line at $\infty$) $z= 0$. 

    Why? $C \cap L$ contains just one point $[0 : 1 : 0]$. Moreover, $\deg \omega_C = 0$. Upshot, $\alpha$ is regular on all of $C$. 

    Next, we talk about which $a_{-1}$ and $b_{-1}$ can occur. We drew the picture of a genus $1$ torus with points $p$ and $q$. We look at $\alpha f$ the meromorphic differential. Then we take contractible opens around $p$ and $q$. 

    By the Residue Theorem, we have 
    \[ 0 = \Res_p(\alpha f) + \Res_q(\alpha f) = \frac{a_{-1}}{3} + \frac{b_{-1}}{-3}\]
    This implies $a_{-1} = b_{-1}$. 

    In general, if $D = \{p_0, \cdots, p_d\} \subset C$ is a divisor on a smooth curve. $\alpha \in H^0(\omega_C)$ and $f \in H^0(\OO_C(D))$. These imply, 
    \[ \sum \Res_{p_i}(\alpha f) = 0\]
    This condition is trivial if and only if $\alpha \in H^0(\omega_w(-D))$. Thus, 
    \[ \dim H^0(\OO_C(D)) \leq d + 1 - [\dim H^0(\omega_C) - \dim H^0(\omega_C(-D))] \]
    That is, 
    \[ h^0(\OO_C(D)) = h^0(\omega_C(-D)) \leq d + 1 - h^0(\omega_C)\]
    Serre Duality says that this is an equality and the terms in this expression agree with the Riemann-Roch theorem. 
\end{example}

From projective to proper. We have seen a number of things we have shown are true for projective morphisms, but claimed they still work for proper morphisms. We are going to cover some lemmas that make this possible. 

The first is Chow's lemma. 
\begin{lemma}
    Let $\pi: X \to \Spec A$ be proper and $A$ noetherian. Then there exists $\mu: X' \to X$ surjective + projective + birational such that $\pi \circ \mu$ is projective. 
\end{lemma}

\begin{proof}[Idea.]
    \bbni
    \begin{enumerate}
        \item Cover $X$ by affines. 
        \item Compactify each affine to a projective $A$-scheme. 
        \item Consider graph of the rational map to their product. 
    \end{enumerate}
\end{proof}


\subsection{Grothendieck's Coherence Theorem}
Let $\pi: X \to Y$ proper morphism of locally noetherian schemes. Let $F$ be coeherent over $X$. Then $R^i \pi_* F$ is coherent.

\begin{proof}[Idea.]
    Reduce to $Y = \Spec A$ and apply Chow's lemma. 
    \[\xymatrix{
        X' \ar[r]^\mu \ar[d]_{\pi \circ \mu} & X \ar[ld]^\pi \\
        \Spec A
    }\]
    Then approximate $F$ by $\mu_* \mu^* F$. Then, 
    \[ 0 \to S \to F \to \mu_*\mu^*F \to Q \to 0\]
    where if $S$ and $Q$ are supported on something of smaller dimension. Then, $R^i \pi_*$ coherent by induction. 

    Let $G$ be $\mu^* F$. Then, use the Leray spectral sequence, 
    \begin{align*}
        R^p \pi_*(R^q \mu_* G) \implies R^{p+q}(\pi \circ \mu)_* G
    \end{align*}
\end{proof}
