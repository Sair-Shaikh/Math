\begin{proposition}[W 1.7]
    Suppose $D \subset \R^2$ is open and connected and $u: D \subseteq \R^2 \to \R$ is such that $u_x \equiv 0 \equiv u_y$. Then $u$ is constant. 
\end{proposition}

\begin{definition}
    $F: D \subseteq \R^2 \to \R^2$ is differentiable at a interior point $z_0 \in D$ if there is a linear map $T: \R^2 \to \R^2$ such that: 
    \[ \lim_{z \to 0} \frac{1}{||z||} ||F(z+z_0) - F(z_0) -T(z)|| = 0 \]
\end{definition}

\begin{remark}
    If $F(z) = (u(x, y), v(x, y))$ is differentiable at $z_0$, then $u$ and $v$ have first partials at $z_0$ and the matrix of $T$ with respect to the standard basis is:
    \[ [T]_\sigma = \begin{pmatrix}
        u_x(z_0) & u_y(z_0) \\
        v_x(z_0) & v_y(z_0)
    \end{pmatrix}\]
    The converse fails! W Ex. 1.2.2.
\end{remark}


\begin{theorem}[W 1.8]
    Suppose that $F : D \subset \R^2 \to \R^2$ is given by $F(x, y) = (u(x,y), v(x,y))$ and $D_r(z_0) \subset D$. If the first partials of $u$ and $v$ exist in $D$ and are continous at $z_0$ then $F$ is differentiable at $z_0$. 
\end{theorem}

If the partials are defined in the neighborhood of $z_0$, it makes sense to talk about their partials, giving us the 2nd derivatives, i.e. $u_{xx}, u_{xy}, u_{yx}, u_{yy}$. 
\begin{theorem}[W 1.9, Clauraut]
    If $u_{xy}$ and $u_{yx}$ are continous at $z_0$, then $u_{xy} = u_{yx}$. 
\end{theorem}

\subsection{Sequences and Series}
Recall that if $(a_n)_{n=1}^\infty$ is a sequence in $\C$ then the series 
\[ \sum_{n=1}^\infty a_n\]
converges if the sequence $(s_n)$ of partial sums converges and then:
\[\sum_{n=1}^\infty a_n = \lim_{n\to \infty} s_n \]
We say the series converges absolutely if $\sum_{n=1}^\infty |a_n| < \infty$. 

\begin{lemma}[W 1.2.9]
    An absolutely convergent series is convergent.
\end{lemma}
\begin{proof}
    Suppose $\sum_{n=1}^\infty |a_n| < \infty$ is absolutely convergent. Then given $\epsilon > 0$, $\exists N$ such that $n \geq N \implies \sum_{n=N}^\infty |a_n| < \epsilon$. Now if $m > n \geq N$, we have: 
    \[ |s_m - s_n| \leq \sum_{k=n+1}^m | a_k | \leq \sum_{k = N}^\infty |a_k| < \epsilon\]
    Thus, $(s_n)$ is Cauchy and by completeness of $\C$, it converges. 
\end{proof}

\begin{theorem}[W 1.1.30]
    Let $(a_n) \subset \C$. Then,
    \begin{enumerate}
        \item If $|a_n| < b_n$ and $\sum_{n=1}^\infty b_n < \infty$, then the series $\sum a_n$ is absolutely convergent. 
        \item If $\lim_{n \to \infty} \left|\frac{a_{n+1}}{a_n}\right| = L$, then $\sum_{n=1}^\infty a_n$ is absolutely convergent if $L < 1$ and diverges if $L > 1$. 
        \item Let $L = \limsup |a_n|^{1/n}$. If $L < 1$, then $\sum a_n$ converges absolutely and diverges if $L > 1$. 
    \end{enumerate}
\end{theorem}

Recall, the definition of a $\limsup$. For $(a_n)$, let $\alpha_n = \sup \{ a_k : k \geq n\}$. This is monotonic decreasing. Then, $\limsup_n a_n = \lim_n \sup_{k\geq n} a_k$. 

Recall, if $(f_n)$ is a sequence of $\C$ valued functions on $D \subseteq \C$ then we say $(f_n)$ converges pointwise on $D$ to some function $f: D \subset \C \to \C$ if:
\[ \lim_{n\to \infty} f_n(z) = f(z) \qquad \forall z \in D\]
We say $(f_n)$ converges uniformly to $f$ on $D$ if $\forall \epsilon > 0$, there exists $N$ such that $n \geq N$ implies
\[ |f_n(z)-f(z)| < \epsilon \qquad \forall z \in D\]

We use the same terminology for the series $\sum_{n=1}^\infty f_n$ by condering the appropriate partial sums. 

\begin{theorem}[W 1.31, Weierstrass M-test]
    Suppose $D \subseteq \C$ and $(f_n)$ is a sequence of $\C$-valued functions on $D$ such that $|f_n(z)| \leq M_n$ for all $z \in D$. If $\sum_{n=1}^\infty M_n < \infty$, then $\sum_{n=1}^\infty f_n$ converges uniformly on $D$ to some $f: D \subseteq \C \to \C$. 
\end{theorem}

\begin{proof}
    Note that $\forall z \in D$, $\sum_{n=1}^\infty f_n(z)$ converges to some $f(z) \in \C$ by comparision test. If $\epsilon > 0$, then $\exists N$ such that $n \geq N \implies \sum_{n=N}^\infty M_n < \epsilon$. If $n \geq N$, then
    \begin{align*}
        |f(z) - s_n(z)| &= |f(z) - \sum_{k=1}^n f_k(z)| \\
        &= \left|\sum_{k = n+1}^\infty f_k(z)\right| \\
        &\leq \sum_{k=n+1}^\infty |f_k(z)| \\
        &\leq \sum_{k=n+1}^\infty M_k < \epsilon
    \end{align*}
\end{proof}

\begin{definition}
    Let $(a_n)_{n=0}^\infty \subset \C$ and $z_0 \in \C$, then the series: 
    \[ \sum_{n=0}^\infty a_n(z-z_0)^n = a_0 + a_1(z-z_0) + \cdots \]
    is called a power series centered at $z_0$ with coefficients $a_k$. 
\end{definition}

\begin{theorem}[W 1.34]
    Consider  
    \[ P : \sum_{n=0}^\infty a_n(z-z_0)^n = a_0 + a_1(z-z_0) + \cdots \]
    Then, there exists $R \in [0, \infty]$ such that: 
    \begin{enumerate}
        \item $P$ converges absolutely if $|z-z_0| < R$
        \item $P$ diverges if $|z-z_0| > R$
        \item and if $0 < r < R$, then $P$ converges uniformly on $D_r(z_0)$. 
    \end{enumerate}
    Furthermore,
    \[ \frac{1}{R} = \limsup_n|a_n|^{1/n} \]
\end{theorem}

\begin{proof}
    Can assume $z_0 = 0$ without loss of generality. $P$ always converges at $z = 0$. If $P$ converges only at $z = 0$, then take $R = 0$ so that $(a)$ and $(b)$ hold and $(c)$ is vacuous. 
    
    Otherwise, suppose the $P$ converges at another point $z_1$ with $|z_1| > 0$. Then, $\exists M$ such that:
    \[ |a_nz_1^n| \leq M \qquad \forall n\]
    If $|z| < |z_1|$, then, 
    \begin{align*}
        |a_nz^n| = |a_nz_1^n| \left|\frac{z}{z_1}^n\right| \leq M \left|\frac{z}{z_1}^n\right|
    \end{align*}
    and $P$ converges absolutely at $z$ by the comparison test. 

    Let $R = \sup \{ |z| : P \text{ converges at } z\}$. If $|z| < R$, then $\exists z$ such that $P$ converges at $z$ and $|z| < |z_1|$. But then, $P$ converges absolutely at $z$. If $|z| > R$, we know that $P$ diverges at $z$. Thus, $(a)$ and $(b)$ hold with $R$. 

    If $0 < r < R$, then $P$ converges absolutely at $z = r$, and we can let $M_n = |a_n|r^n$ and then $P$ converges uniformly on $D_r(0)$ and (c) holds.
    Now let,
    \[ R_0^{-1} = \limsup_n |a_n|^{1/n}\]
    If $0 < |z| < R_0$, then
    \[ \limsup_n |a_nz^n|^{1/n} = R_0^{-1}|z| < 1\]
    Thus, $P$ converges absolutely at $z$. 
    
    If $|z| > R_0$, then, 
    \[  \limsup_n |a_nz^n|^{1/n} = R_0^{-1} |z| > 1\]
    Thus, $P$ diverges. 

    Thus, $R_0 = R$. 
\end{proof}

Recall, that a subset $D \subset \C$ is connected if it does not admit a non-trivial partition into relatively open subsets.
\begin{remark}
    $A \subset \C$ is connected if given open sets $U, V \in \C$ such that $\{ A \cap U, A \cap V \}$ is a partition and $A \cap U$ is not empty, then $A \cap V$ is empty. 
\end{remark} 

\begin{definition}
    We call a connected, open subset $D \subset \C$ a domain. 
\end{definition}

If $E \subset \C$ and $z \in E$, then 
\[ E(z) := \bigcup \{ A \subset \C : z \in A \subset E \text{ and } A \text{ connected} \}\]
Then $E(z)$ is connected and if $E(z) \cap E(w) \neq 0$ then, $E(z) = E(w)$. These form a partition of $A$ and are called the connected components of $E$. 

Since disks $D_r(z_0)$ are connected, it follows that the connected components of an open set $U$ are open. (W Ex. 2.1.2). That is, every open subset is a countable union of domains. 

We say $E \subset \C$ is path connected if given $z, w \in E$, there is a continous function: 
\[ \gamma: [0, 1] \to E \]
such that $\gamma(0) = z$ and $\gamma(1) = w$. Since $[0, 1]$ is connected, it follows that path-connected sets are connected. 

For an connected open set, the converse holds (because disks are path-connected).

\section{Complex Differentiability}

\begin{definition}
    Suppose that $f: D \subset \C \to \C$ and that $z_0 \in D$ is an interior point. Then $f$ is complex differentiable at $z_0$ if 
    \[ f'(z_0) = \lim_{h\to 0} \frac{f(z_0+h) - f(z_0)}{h}\]
    exists. Then we call the $f'(z_0)$ the derivative of $f$ at $z_0$. If $D$ is open and $f'(z)$ exists for all $z \in D$, then we say that $f$ is holomorphic on $D$ and write $H(D)$ for the set of all holomorphic functions on $D$. If $f \in H(\C)$, then we say $f$ is entire. 
\end{definition}

\begin{remark}
    If $f(z) = z^n$ for $n \in \N$, then $f'(z) = nz^{n-1}$ with the usual proof. Similarly, the sum, product, quotient, and chain rules all apply with the usual proofs. 
\end{remark}

\begin{remark}
    Polynomials are entire. Rational functions are holomorphic on their natural domains.
\end{remark}

\begin{example}
    Let $f(z) = |z|^2$. Thus, if $z = x+iy$, $f(z) = x^2 + y^2$. Consider $\C = \R^2$, this function is $(x, y) \mapsto (x^2 + y^2, 0)$. Recall, that $|z|^2 = z\overline{z}$. Thus, 
    \begin{align*}
        f'(z) &= \lim_{h \to 0} \frac{|z+h|^2 - |z|^2}{h} \\
        &= \lim_{h \to 0} \frac{z\overline h + \overline z h + h \overline h}{h}\\
        &= \overline{z} + z \lim_{h \to 0 }\frac{\overline h}{h}
    \end{align*}
    Suppose $f'(z)$ exists and $h = x+iy$. Then, 
    \begin{align*}
        f'(x) &= \overline{z} + z\lim_{x+iy \to 0} \frac{x+iy}{x+iy} \\
        &= \overline{z} + z \lim_{x \to 0} \frac{x}{x} = \overline{z} + z
    \end{align*}
    Simultaneously, 
    \[ f'(z) = \overline{z} + z\lim_{y\to 0} \frac{-y}{y} = \overline{z} - z\]
    These force $z = 0$ and $f'(0) = 0$. 
\end{example}

\begin{theorem}[W 2.4]
    Suppose $r > 0$ and
    \[ f(z) = \sum_{n = 0}^\infty a_n(z-z_0)^n\]
    for $z \in D_r(z_0)$.  Then $f \in H(D_r(z_0))$ and 
    \[ f'(z) = \sum_{n=1}^\infty na_n(z-z_0)^{n-1}\]
\end{theorem}

\begin{proof}
    Since $\lim_{n \to \infty} n^{1/n} = 1$, the radius of convergence of the function and its derivative are equal. Thus, $\forall z \in D_r(z_0)$ it suffices to see that $f'(z)$ exists and equals $g(z) = \sum_{n=1}^\infty na_n(z-z_0)^{n-1}$. 

    Using a change of variables, we can assume $z_0 = 0$. Fix $z \in D_r(0)$ and $p$ such that $0 \leq |z| < p < r$. If $w \neq z$, then, 
    \begin{align*}
        \frac{f(w)-f(z)}{w-z} -g(z) &= \sum_{n = 1}^\infty \left(\frac{a_nw^n - a_nz^n}{w-z} - na_nz^{n-1} \right)\\
        &= \sum_{n=1}^\infty a_nA_n
    \end{align*}
    where $A_1 = 0$ and for $n \geq 2$, 
    \begin{align*}
        A_n &= \frac{w^n -z^n}{w-z} - nz^{n-1} \\
        &= \sum_{k=0}^{n-1} w^{n-1-k}z^k - nz^{n-1}
    \end{align*}
    We claim this equals, 
    \[ (w-z) \sum_{k=1}^{n-1} kz^{k-1}w^{n-k-1}\]
    To see this, expand it out. 

    Thus if $|w| < p$ and $n \geq 2$, then 
    \begin{align*}
        |A_n| &\leq |w-z| \sum_{k=1}^{n-1} kp^{n-2} \\
        &= |w-z| p^{n-2}\frac{n(n-1)}{2}
    \end{align*}
    Thus if $|w| < p$, 
    \begin{align*}
        \left| \frac{f(w) - f(z)}{w-z} -g(z) \right| \leq |w-z| \sum_{n=2}^{\infty} n^2 |a_n| p^{n-2}
    \end{align*}
    But, we also have
    \begin{align*}
        \limsup_n |a_n|^{1/n} = \frac{1}{R} \qquad p < r \leq R
    \end{align*}
    This implies, 
    \[ \sum_{n=2}^\infty n^2 |a_n|p^{n-2}\]
    converges to some $M > 0$. 

    Therefore, $|w| < p$, then 
    \begin{align*}
        \left| \frac{f(w) -f(z)}{w-z} - g(z)\right| \leq |w-z|M
    \end{align*}
    Thus $f'(z)$ exists and equals $g(z)$. 
\end{proof}

Now we give an alternative characterization of holomorphic. Given $f: D \subset \C \to \C$, 
\[ f(x+iy) = u(x,y) + iv(x,y)\]
where $u,v: D \subset \R^2 \to \R$. The companion function is $F(x,y) = (u(x,y), v(x,y))$. 

\begin{example}
    For instance, take
    \[ f(x+iy) = (x+iy)^3 = x^3 -3xy^2 + i(3x^2y-y^3)\]
    So $u = x^3 -3xy^2$ and $v = 3x^2y-y^3$. 
\end{example}


\begin{definition}
    We say $f(x+iy) = u(x,y) + iv(x,y)$ satisfies the Cauchy-Riemann equations at $z_0$ if 
    \[ u_x(z_0) = v_y(z_0) \qquad u_y(z_0) = -v_x(z_0)\]
    Note that, 
    \[ f_x(z_0) = \lim_{\substack{x\to 0 \\ x \in \R}} \frac{f(z_0+x)-f(z_0)}{x} = u_x(z_0) +iv_x(z_0)\]
    and similarly for $f_y$. Using this notation, these become
    \[ f_x(z_0) = - if_y(z_0) \]

\end{definition}

\begin{theorem}
    If $f(x+iy) = u(x,y) + iv(x,y)$ and $F$ is the companion function. Assume $z_0$ is an interior point of $D$. Then if $f$ is complex differentiable at $z_0$ if and only if $F$ is differentiable at $z_0$ and the Cauchy-Riemann equations hold at $z_0$. In the case $f'(z_0)$ exists, then, 
    \[ f'(z_0) = f_x(z_0) = -if_y(z_0)\]
\end{theorem}

\begin{proof}
    Suppose $f'(z_0)$ does exist. Then, by definition,
    \begin{align*}
        f'(z_0) &= \lim_{x+iy \to 0} \frac{f(z_0)+x+iy - f(z_0)}{x+iy} \\
        &= \lim_{x \to 0} \frac{f(z_0+x)}{x} = f_x(z_0)  = f_x(z_0)\\
        &= \lim_{y \to 0} \frac{f(z_0 + iy)-f(z_0)}{iy} = -if_y(z_0)
    \end{align*}
    The rest is Exercise 2.2.5 on the homework. 
\end{proof}

\begin{corollary}

    \[ f'(x+iy) = f_x(x+iy) = e^x\cos(y) + ie^x\sin(y) = f(x+iy)\]
\end{proof}

\begin{remark}
    If $f(z) = \exp(z)$ then $f$ has derivatives of all orders. Thus, it has a Taylor series at $z_0 = 0$, 
    \[ \sum_{n=0}^\infty \frac{z^n}{n!} \]
\end{remark}

\begin{remark}
    A taylor series, generally, does not have to have anything with the original function. It could not converge outside of the center, it can converge but not agree with the original function except at the center, etc, in the real case. This won't happen to us in the complex world. 
\end{remark}

\begin{proposition}
    Suppose $D$ is a domain and $f \in H(D)$. If $f'(z) = 0$ for all $z \in D$, then $z$ is constant. 
\end{proposition}
\begin{proof}
    Let $f(z) = u(z) + iv(z)$. Since $f'(z) = f_x(z) = -if_y(z)$, this means, 
    \[ u_x \equiv u_y \equiv v_x \equiv v_y \equiv 0\]
    As the domain is connected, $u$ and $v$ are constant, hence $f$ is constant. 
\end{proof}

\begin{lemma}[2.19]
    Suppose $f$ is a branch of $\log(z)$ on a domain $D$. Then $\Re(f(z)) = \ln(|z|)$, $f \in H(D)$, and $f'(z) = \frac{1}{z}$ for all $z \in D$. 
\end{lemma}
\begin{proof}
    By definition, $f$ is continuous and $\exp(f(z)) = z$ for all $\forall z \in D$. Taking modulus on both sides, we get, 
    \begin{align*}
        &|z| = |\exp(f(z))| = \exp(\Re(f(z))) \\
        \implies & \ln(|z|) = \Re(f(z))
    \end{align*}
    Now let $z_0 \in D$. We want to show, 
    \[ \lim_{z \to z_0} \frac{f(z) - f(z_0)}{z-z_0} = \frac{1}{z_0} \]
    Let $w_0 = f(z_0)$ and fix $\epsilon > 0$. Since, 
    \begin{align*}
        \lim_{w \to w_0} \frac{e^w - e^{w_0}}{w-w_0} = e^{w_0} = z_0
    \end{align*}
    Since $e^{w_0} \neq 0$, then there exists $\delta' > 0$, such that,
    \[ 0 \leq |w-w_0| < \delta' \implies e^w - e^{w_0} \neq 0 \]
    and 
    \begin{align*}
        \left| \frac{w-w_0}{e^w - e^{w_0} - \frac{1}{z_0}}\right| < \epsilon
    \end{align*}
    Since $f$ is continuous, $\exists \delta > 0$ such that $0 \leq |z-z_0| < \delta$, we have,
    \[ |f(w) -f(w_0)| < \delta' \]
    Now if we write $w = f(z)$, then, 
    \begin{align*}
        \left| \frac{f(z)-f(z_0)}{z-z_0} - \frac{1}{z_0}\right| \\
        &= \left| \frac{w-w_0}{e^w-e^{w_0}} - \frac{1}{z_0}\right| < \epsilon
    \end{align*}
    Since $0 \leq |w-w_0| < \delta'$, 
    \[ f'(z_0) = \frac{1}{z_0}\]
\end{proof}

\begin{definition}
    Suppose that $u: D \subseteq \R^2 \to \R$ has continous 2nd partials in $D$. Then $u$ is harmonic on $D$ if, 
    \[\Delta u(x,y) = \frac{\d u}{\d x^2}(x,y) + \frac{\d u}{\d y^2}(x,y)  = 0\]
    That is, 
    \[ u_{xx}(x,y) + u_{yy}(x,y) = 0\]
\end{definition}

\begin{proposition}
    Suppose $D$ is a domain and that $f(x+iy) = u(x,y) + iv(x,y)$ is in $H(D)$. If $u$ and $V$ have continuous 2nd partials, then $u$ and $v$ are harmonic in $D$. 
\end{proposition}

\begin{proof}
    We can use Clairaut's Theorem. 
    \begin{align*}
        u_{xx} = \frac{\d}{\d x} u_x = \frac{\d}{\d x} v_y = \frac{\d}{\d y} v_x = -\frac{\d}{\d y} u_y = u_{yy}
    \end{align*}
    Note that $v = \Re(-if)$ thus this proof works for the imaginary part too. 
\end{proof}

\begin{definition}
    If $u$ is harmonic in $D$, then we call $v$ a harmonic conjugate in $D$ if 
    \[ f(z) = u(z) + iv(z)\]
    is holomorphic in $D$. 
\end{definition}

\begin{lemma}
    Suppose that $v$ and $w$ are harmonic conjugates for $u$ in a domain $D$. Then $\exists r \in \R$ such that, 
    \[ w(z) = v(z) + r \qquad \forall z \in D\] 
    By assumption, $f = u+iv$ and $g = u+iw$ are in $H(D)$. Thus, 
    \[ h = -i(f-g) \in H(D)\]
    but this is real-valued, hence constant. 
\end{lemma}

\begin{corollary}
    There is no branch of the logarithm in $\C \setminus \{0\}$. 
\end{corollary}
\begin{proof}[Sketch.]
    We have $\Log(z) = \ln(|z|) + i \Arg(z)$ is holomorphic in $D^* = \C \setminus (-\infty, 0]$. We can take another branch, 
    \[ \LL_0(z) = \ln(|z|) + i\arg_0(z)\]
    where $\arg_0 \in (0, 2\pi]$. This is holomorphic on $D_0^* = \C \setminus [0, \infty)$. Thus, $u(z) = \ln(|z|)$ is harmonic everywhere on $\C \setminus \{0\}$. 

    But $u$ has no harmonic conjugate on $\C \setminus \{0\}$. If $v$ was such a harmonic conjugate, it would be a harmonic conjugate to $u$ in $D^* = \C \setminus (-\infty, 0]$. 

    That is, $v(z) = \Arg(z) + r$ for all $z \in D^*$. 
\end{proof}

\begin{definition}
    If $X$ is a topological space, then a curve in $X$ is a continuous function $\gamma: [a, b] \subset \R \to X$. We call $[a,b]$ the parameter space and write $\gamma^*$ for the image $\gamma([a,b]) \subset X$. If $\gamma(a) = \gamma(b)$, we call $\gamma$ closed and we also have $\gamma(s) \neq \gamma(t)$ when $a \leq s < t < b$, then we say $\gamma$ is a simple closed curve. 
\end{definition}

We're almost always going to have $X \subset \C$. Then we write, 
\[ \gamma(t) = x(t) + iy(t) \]
We let
\[ \gamma'(t) = \lim_{h \to 0} \frac{1}{h} (\gamma(t+h)-\gamma(t))\]
if it exists. This happens if and only if $x'(t)$ and $y'(t)$ exist. Then, 
\[ \gamma'(t) = x'(t) + y'(t) \]

We say that $\gamma$ is smooth if $\gamma'(t)$ exists and is non-zero for all $t \in [a,b]$. 

\begin{example}
    Consider $\gamma_1(t) = t+it^2$ if $t \in [-1, 1]$ and $\gamma_2(t) = t^3+it^6$ for $t \in [-1, 1]$. The first one is smooth and the second one is not. 
\end{example}

\begin{example}
    Take $\gamma(t) = t^3 + it^2$ for $t \in [-1,1]$. This is $y = x^{\frac{2}{3}}$. 
\end{example}

\begin{theorem}[Jordan Curve Theorem.]
    If $\gamma$ is a Jordan curve, then $\C \setminus \gamma^*$ consists of two domains, each of which has $\gamma^*$ as its boundary. 
\end{theorem}

\begin{remark}
    Since $\gamma^*$ is compact, it is contained in some $D_R(0)$ and hence exactly one of the components of $\C \setminus \gamma^*$ is bounded and we call that the interior of $\gamma$. 
\end{remark}

\begin{definition}
    A curve $\gamma: [a,b] \to \C$ is called a path or a contour if there is a partition 
    \[ \mathcal P = \{ a = t_0 < t_1 < \cdots < t_n = b\}\]
    of $[a,b]$ such that $\gamma'(t)$ is continuous on $[t_{k=1}, t_k]$ for $1 \leq k \leq n$. If $D \subset \C$, then we say $\gamma$ is a path in $D$ if $\gamma^* \subset D$. 
\end{definition}

\begin{definition}
    Let $\gamma: [a, b] \to \C$ be a curve. Say $\gamma(t) = x(t)+iy(t)$. We define, 
    \[ \int_{a}^b \gamma(t)dt = \int_{a}^b x(t)dt + i\int_a^b y(t)dt\]    
    If $\gamma$ is continuous, we have, 
    \[ \int_{a}^b \gamma'(t)dt = x(b)-x(a) +i(y(b)-y(a))\]    
\end{definition}
\begin{remark}
    You can do the whole Riemann sums business and then use the completeness of $\C$ to show this, but we just skip that by defining this. 
\end{remark}

\begin{proposition}
    Suppose $\gamma$ and $\sigma$ are curves defined on $[a,b]$. Then, 
    \begin{enumerate}
        \item[a)] $\int_a^b (\gamma(t) + \sigma(t))  = \int_a^b \gamma(t)dt + \int_a^b \sigma(t)dt$.
        \item[b)] If $\alpha \in \C$, then $\int_a^b \alpha\gamma(t)dt = \alpha \int \gamma(t)dt$. 
        \item[c)] $\left| \int \gamma(t)dt\right|  \leq \int|\gamma(t)|dt$
    \end{enumerate}
\end{proposition}
\begin{proof}
    We are not allowed to use the word trivial. Thus, we say that the proof for a) is straightforward. Similarly, b) is routine if $\alpha \in \R$. If $\alpha = a+ib$, we can multiply it out and use the defintion. For c), we let $\tau$ be a unimodular constant such that,
    \[ |\int \gamma(t)dt| = \tau \int \gamma(t)dt \]
    Now, 
    \begin{align*}
        \left|\int \gamma(t) dt \right| &= \int \tau \gamma(t)dt \\
        &= \Re( \int \tau \gamma(t)dt) \\ 
        &= \int \Re(\tau\gamma(t))dt \\
        &\leq \int |\tau\gamma(t)|dt \\
        &= \int |\gamma(t)| dt
    \end{align*}
\end{proof}

\begin{definition}
    If $\gamma: [a,b] \to \C$ is a path, then the opposite path, $-\gamma$ is $-\gamma: [a,b] \to \C$ where 
    \[ -\gamma(t) = \gamma(b+a-t) \]
\end{definition}

\begin{definition}
    If $z, w \in \C$, then $[z, w]$ is the path $\gamma: [0,1] \to \C$ such that $\gamma(t) = z+t(w-z)$. 
\end{definition}

\begin{remark}
    Note that $-[z, w] = [w, z]$. 
\end{remark}

\begin{remark}
    If $\gamma: [a,b] \to \C$ is a path and $\gamma(t) = x(t) + iy(t)$ then there exists $\mathscr P = \{a = t_0 < t_1 < \cdots < t_n = b\}$ such that $\gamma'(t)$ is continuous on $[t_{k-1}, t_k]$. Then $x'$ and $y'$ are Riemann integrable on $[a,b]$. 
\end{remark}

\begin{definition}
    If $\gamma: [a,b] \to \C$ is any curve, then we can define, 
    \[ \int_a^b \sigma(t)\gamma'(t)dt = \sum_{k=1}^n \int_{t_{k-1}}^{t_k} \sigma(t)\gamma'(t)dt \]
    Similarly, we'll write, 
    \[ \int |\gamma'(t)|dt = \sum_{k=1}^n \int_{t_{k-1}}^{t_k} |\gamma'(t)|dt\]
\end{definition}

\begin{definition}
    If $\gamma: [a,b] \to \C$ is a path and if $f \in C(\gamma^*)$. Then, the path integral (or contour integral) of $f$ along $\gamma$ is, 
    \[ \int_\gamma f(z)dz = \int_a^b f(\gamma_)\gamma'(t)dt \]
\end{definition}

\begin{remark}
    We call $\gamma: [0, 2\pi] \to \C$ given by 
    \[ \gamma(t) = a + re^{it}\]
    the postively oriented circle of radius $r$ centered at $a$. 
\end{remark}

\begin{lemma}[Homer's Cauchy Theorem]
    Let $\gamma(t) = a+re^{it}$ with $r > 0$ and $t \in [0, 2\pi]$. Then if $n \in \Z$, we have, 
    \begin{align*}
        \int_\gamma (z-a)^ndz = \begin{cases}
            0 \qquad if n \neq -1 \\
            2\pi i \qquad n = -1
        \end{cases}
    \end{align*}
\end{lemma}
\begin{proof}
    If $\sigma(t) = e^{\alpha t}$ with $\alpha \in \C$, then, 
    \[ \sigma'(t) = \alpha e^\alpha t\]
    Thus, $\gamma'(t) = rie^{it}$. If $n \neq -1$, then, 
    \begin{align*}
        \int_\gamma (z-a)^ndz &= \int_\gamma (re^{it})^n rie^{it} dt \\
        &= r^{n+1}i\int_0^{2\pi} e^{i(n+1)t} \\
        &= ir^{n+1} \frac{e^{i{n+1}t}}{i(n+1)}\big|_0^{2\pi} = 0
    \end{align*}
\end{proof}
