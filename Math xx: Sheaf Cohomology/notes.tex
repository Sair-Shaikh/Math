\section{Sheaf Cohomology:}
Recall that the global sections functor, $\Gamma$ is left-exact. 

Our goal is to take $X$, a separated, quasi-compact $A$-scheme and $F$ a quasi-coherent sheaf of $\OO_X$ modules, to produce sheaf cohomology groups $H(X, F)$ satisfying: 
\begin{enumerate}
    \item $H^i(X, -): QCoh \to Mod_A$ are functors. 
    \item $H^0(X, -) = \Gamma(X, -)$  as functors.
    \item Given exact seqeunce $0 \to F \to G \to H \to 0$, then we want the long exact sequence: 
    \[ 0 \to H^0(X, F) \to H^0(X, G) \to H^0(X, H) \to H^1(X, F) \to  H^1(X, G) \cdots \] 
    \item Given $\pi: X \to Y$ and $F$ a qcoh on $X$, you will get $H^i(Y, \pi_*(F)) \to H^i(X, F)$ extending the usual pullback map on global sections. 
    \item The map in (4) is an isomoprhism if $\pi$ is affine. 
    \item If $X$ covered by $n$ affines, then $H^1(X, F) = 0$ for $i \geq n$. In particular, $\Gamma$ is exact for $X$-affine.
    \item $H^i(X, \colim F_j) = \colim H^1(X, F_j)$. In particular, $H^i(X, F_1 \oplus F_2) = H^1(X, F_1) \oplus H^1(X, F_2)$. 
    \item The cohomology of line bundles on projective space is ... look at the next theorem.
\end{enumerate}

For projective space, we want the following:
\begin{theorem}
    .
    \begin{enumerate}
        \item $H^0(\P_A^n, \OO(m))$ is free of rank $\binom{n+m}{m}$ if $m \geq 0$. 
        \item $H^n(\P^n_A, \OO(m))$ is free of rank $\binom{-m-1}{-n-m-1}$ if $m \leq -n-1$
        \item $H^i(\P^n_A, \OO(m)) = 0$ in all other cases. 
    \end{enumerate}
\end{theorem}

Observe now the following: 
\begin{enumerate}
    \item These are all finitely generated $A$-modules. This is true for any coherent sheaf on a proper $A$-scheme. 
    \item These vanish for $i > n = \dim(\P^n_A)$. 
    \item The top cohomology, $H^n(\P^n_A, \OO(m))$ is $1$ dimensional for $m = -n-1$ and induces a perfect pairing:
    \[ H^i(\P_k^n, \OO(m)) \times H^{n-i}(\P^n_k, \OO(-n-1-m)) \to H^n(\P^n_k, \OO(-n-1)) = k\]
    This is a special case of Serre duality. $\OO(-n-1)$ is a special case of a ``dualizing sheaf''. 
    \item Higher cohomology vanishes for $m > -n-1$. This is a special case of the Kodaira vanishing theorem.
    \item $\Sigma_i (-1)^i h^i(\P_k^n, \OO(m))$ is a polynomial in $m$. This is called the Hilbert polynomial of projective space. 
\end{enumerate}

\begin{theorem}
    Let $X$ be projective over $A$ noetherian and $F$ a coherent sheaf on $X$. Then, 
    \begin{enumerate}
        \item The $H^i(X, F)$ are finitely generated $A$-modules
        \item The $H^i(X, F(m))$ vanish for $i > 0$ and $m >> 0$. Note: $F(m) = F \otimes \OO(m)$. 
    \end{enumerate}
\end{theorem}

\begin{proof}
    Step 1: Reduce to $X = \P_A^n$. For $\pi : X \hookrightarrow \P_A^n$, we have:
    \[ H^i(X, F(m)) =  H^1(\P^n_A, \pi_*(F(m)))\]
    because closed embeddings are affine. So we might as well do this for projective space. 

    Step 2: Present $F$ as: 
    \[ 0 \to G \to \OO(m)^j \to F \to 0 \]
    giving the long exact sequence:
    \[ 
    \xymatrix{
    0 \ar[r] & H^0(G) \ar[r] & H^0(\OO(m)^j) \ar[r] & H^0(F) \ar[lld] \\
    & H^1(G) \ar[r] & H^1(\OO(m)^j) \ar[r] & H^1(F) \ar[lld] \\
    & H^2(G) \ar[r] & H^2(\OO(m)^j) \ar[r] & H^2(F) \ar[lld] \\
    & \vdots \ar[r] & \vdots \ar[r] & \vdots \ar[lld] \\
    & H^{n-1}(G) \ar[r] & H^{n-1}(\OO(m)^j) \ar[r] & H^{n-1}(F) \ar[lld] \\
    & H^n(G) \ar[r] & H^n(\OO(m)^j) \ar[r] & H^n(F) \ar[r] & 0
    } \]


    Step 3: Backwards induction. 
    \begin{enumerate}
        \item $H^n(\OO(m)^j)$ finitely generated implies $H^n(F)$ finitely generated for all $F$ implies $H^n(G)$ finitely generated implies $H^{n-1}(F)$ finitely generated for all $F$. 
        \item Twist by $\OO(N)$ for $N >> 0$. Then, 
        \[ H^n(\OO(m+N)^j)= 0 \implies H^n(F(N)) = 0 \]
        for all $F$ and $N$ sufficiently large. This implies, that $H^n(G(N)) = 0$ for $N >>_F 0$ which implies $H^{n-1}(F(N)) = 0$. 
    \end{enumerate}
    In particular, we showed that $\Gamma(X, F)$ is a finitely generated $A$-module.
\end{proof}

\begin{corollary}
    Let $Y$ is locally noetherian. Then $\pi: X \to Y$ is finite if and only if it is affine and projective. 
\end{corollary}
