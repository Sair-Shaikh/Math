\section{Motivation:}

I will begin by stating the Weil Conjectures (there before class):

\begin{theorem}
    If $X$ is a non-singular, geometrically connected, $n$-dimensional projective variety over $\F_q$. Define:
    \[ Z(t) = \exp\left(\sum_{m\geq 1} \frac{N_m}{m}t^m\right)\]
    Then, the following hold: 
    \begin{enumerate}
        \item (Rationality) $Z(t)$ is a rational function, i.e. $Z(t) \in \Q(t)$.
        \item (Functional Equation) If $E$ is the Euler characteristic of $X$, then:
        \[ Z(\frac{1}{q^nt}) = \pm q^{nE/2}t^E Z(t) \]
        \item (Riemann Hypothesis) One can write:
        \[ Z(X,t) = \frac{P_1(t)\cdot P_3(t) \cdot P_{2n-1}(t)}{P_0(t)\cdot P_2(t) \cdot P_{2n}(t)} \]
        with $P_0(t) = 1-t$, $P_{2n}(t) = 1-q^nt$ and for $1 \leq 2n-1$, we have $P_i(t) \in \Z[t]$, 
        \[ P_i(t) = \prod_j (1-\alpha_{i,j}t) \] 
        with $|\alpha_{i,j}| = q^{i/2}$.
        \item (Comparison Theorem) If $X$ come from a good reduction mod $p$ of a non-singular projective variety $Y$ defined over a number field in $\C$, then the degree of $P_i$ is the $i$th Betti number of $Y$.
    \end{enumerate}
\end{theorem}
\begin{remark}
    A priori, we do not have a defined Euler characteristic. But once we have rationality, we can define it as the degree of the denominator minus the degree of the numerator. 
\end{remark}

\begin{remark}
    Weil proved these conjectures for curves of arbitrary genus in 1948 and for diagonal hypersurfaces in 1949. This is why he conjectured them.
\end{remark}

In this talk, we will reduce the problem of proving the first two of these to finding a cohomology theory satisfying some axioms, called a Weil Cohomology theory.

To get a hold of the zeta function, we really need to understand $X(\F_q^r)$. Recall, from Henry's talk (initial object in the category of talks),
\begin{theorem}
    Let $X$ be a non-singular projective variety over $\F_q$.  We have the absolute frobenius $\Frob_{q}: X \to X$ given by $\id$ on $X$ and $u \mapsto u^q$ on $\OO_X$. Then, $X(\F_{q^r})$ is in natural bijection with $X(\overline{\F}_q)$ fixed by $\Frob_q^r$. \\
    More geometrically, define $\overline{X} = X \times_{\F_q} \overline{\F}_q$ and $F_q := \Frob_q \times \Id$ on $\overline{X}$. Let $\Delta, \Gamma_r \subset \overline{X} \times \overline{X}$ be the diagonal and the graph of $F^r$, respectively. Then $X(\F_q^r)$ is in bijection with the closed points of $\Gamma_r \cap \Delta$.
\end{theorem}

Thus, we need a way to count the closed points of $\Gamma_r \cap \Delta$. To this end, we do have a result from classical topology, the Lefschetz fixed-point theorem (from Ribhu's talk):
\begin{theorem}
    Let $M$ be an $n$-dimensional oriented compact manifold without boundary and $f: M \to M$ a map with isolated fixed points. Then, 
    \[  (\Gamma_f \cdot \Delta) = \sum_{i = 0}^n (-1)^i \Tr(f_i^*|H^i) \]
\end{theorem}

We recall from Ribhu's talk that the proof of this required a few key ingredients: 
\begin{enumerate}
    \item We converted the problem of counting the intersection number to linear algebra. In particular, for submanifolds $S, N \subset M$, we found forms $\eta_S, \eta_N$ such that:
    \[ (S \cdot N) = \int_M \eta_S \wedge \eta_N \]
    \item We used some nice properties of the cohomology theory, such as Poincare duality and the Kunneth formula to derive the result. 
\end{enumerate}

We want to do something similar in the algebraic geometry setting, where intersection theory is more complicated. So we start by reviewing some properties of intersection theory.

\section{Intersection Theory Interlude:}

Let $X$ be a smooth variety over $k$. 

\begin{definition}
    \begin{enumerate}
        \item Define the groups $Z_r(X)$ are free abelian group generated by closed, irreducible subvarieties of dimension $r$. The elements of $Z_r(X)$ are called algebraic ``$r$-cycles".
        \item Let $V$ be a $(k+1)$ dimensional subvariety of $X \times \P^1$ such that $f: V \to \P^1$ is dominant. For $P \in \P^1$ rational over the ground field, the fiber $f^{-1}(P)$ is isomorphic to a subvariety of $X$. Denote this as $V(P)$. Rational equivalence is the equivalence relation generated by setting: 
        \[ [V(0)]-[V(\infty)] \sim 0\]        
        \item Define the Chow groups $A_r(X)$ are the quotient groups under the rational equivalence relation $\sim_{rat}$. For $Z \in Z_r(X)$, $[Z] \in A_r(X)$ is the class of $Z$. Set $A^i = A_{n-i}$.
        \item There is an intersection product making $A^*(X)$ into a graded ring. This is difficult to define in generality, but is related to the intuitive intersection of subvarieties. \\
        For $W \in A^i(X)$ and $Z \in A^{n-i}(X)$, 
        \[ [W \cap Z] = \sum_{P \in W \cap Z} l(\OO_{W \cap Z, P})[P] \]
    \end{enumerate}
\end{definition}

From morphisms on varieties, we have two fundamental morphisms induced on the Chow groups:

\begin{definition}[Proper pushforward]
    Let $f: X \to Y$ be a proper morphism. For a closed subvariety, $V \subset X$ of dimension $r$, we define: 
    \[f_*: Z_r(X) \to Z_r(Y), \qquad f_*(V) = \deg(V/f(V))[f(V)] \]
    where $\deg(V/f(V))$ is the degree of the extension $[K(V) : K(f(V))]$ if $\dim(f(V)) = \dim(V)$ and $0$ otherwise.
\end{definition}
\begin{remark}
    We need properness so that fibers are compact and $\deg(V/f(V))$ is well-defined. 
\end{remark}

\begin{theorem}
    The pushforward has the following properties: 
    \begin{enumerate}
        \item The pushforward induces a well-defined map $A_r(X) \to A_r(Y)$.
        \item It is (covariant) functorial.
        \item For $\alpha = \sum_P n_P [P]$ a $0$-cycle, define the degree of $\alpha$ by $\deg(\alpha) = \sum_P n_P$. Then, $\deg(\alpha) = f_*(\alpha)$ where $f: X \to \Spec k$. 
    \end{enumerate}
\end{theorem}

\begin{definition}[Flat Pullback] Let $f: X \to Y$ be a flat morphism of relative dimension $r$. Define the pull-back map:
    \[f^*: Z_i(Y) \to Z_{i+r}, \qquad f^*([V]) = [f^{-1}([V])]\]
\end{definition}
\begin{remark}
    We need flat so that the fibers are well-behaved. 
\end{remark}

\begin{theorem}
    The pullback has the following properties:
    \begin{enumerate}
        \item The pushforward induces a well-defined map $A_r(X) \to A_r(Y)$.
        \item It is (contravariant) functorial.
    \end{enumerate}
\end{theorem}

\section{Weil Cohomology Theories:}

% To achieve the goals of a Lefschetz fixed point theorem, we define a Weil cohomology theory to reflect the intersection theory as follows:

\begin{definition}(Weil Cohomology Theory) \\
    A Weil Cohomology theory with coefficients in a char $0$ field $K$ consists of the following data:
    \begin{enumerate}
        \item A contravariant functor $X \to H^*(X) = \bigoplus_i H^i(X)$ from the category of smooth, connected, projective varieties over (algebraically closed) $k$ to the category of graded commutative $K$-algebras. The product of $\alpha, \beta \in H^*(X)$ is denoted $\alpha \cup \beta$. We write $f^* := H(f)$. 
        \item A linear map $\Tr = \Tr_X: H^{2\dim(X)}(X) \to K$.
        \item The cycle class map: $\cl: Z^r(X) \to H^{2r}(X)$ mapping each closed irreducible subvariety $Z \subseteq X$ of codimension $r$ to a cohomology class $\cl(Z)$.
    \end{enumerate}
    On this data, we require the following: 
    \begin{enumerate}
        \item (Fin. dim.) All $H^i(X)$ are finite dimensional vector spaces over $K$. Moreover, $H^i(X) = 0$ except when $0 \leq i \leq 2 \dim(X)$.
        \item (Point) If $X = \Spec k$, then $\cl(X) = 1$ and $\Tr_X(1) = 1$. 
        \item (Kunneth property) For $X, Y$, if we define $p_X:X \times Y \to X$ and $p_Y: X \times Y \to Y$ to be the projections, then the $K$-algebra homomorphism: 
        \[ H^*(X) \otimes_K H^*(Y) \to H^*(X \times Y), \quad \alpha \otimes \beta \mapsto p_X^*(\alpha) \cup p_Y^*(\beta) \] 
        is an isomorphism. 
        \item (Poincaré duality) We have $\Tr: H^{2\dim(X)}(X) \cong K$, and for every $0 \leq i \leq 2 \dim(X)$, the bilinear map: 
        \[ 
            H^i(X) \otimes_K H^{2 \dim(X) -i}(X) \to K, \quad (\alpha, \beta) \mapsto \Tr_X(\alpha \cup \beta)
        \]
        is a perfect pairing.
        \item (Exterior product) For varieties $X, Y$ and every closed irreducible varieties $Z \subseteq X$ and $W \subseteq Y$, we have:
        \[
            \cl(Z \times W) = p^*_X(\cl(Z)) \cup p^*_Y(\cl(W))
        \]
        and further if $\cl(Z) \in H^{2\dim(X)}(X)$ and $\cl(W) \in H^{2\dim(Y)}(Y)$ then: 
        \[  \Tr_{X\times Y}(\cl(Z \times Y)) = \Tr_X(\cl(Z))\Tr_Y(\cl(W)) \] 
        \item (Pushforward) For every (proper) morphism $f: X \to Y$ and for every irreducible closed subvariety $Z \subseteq X$, we have for every $\beta \in H^{2 \dim(Z)}(X)$:
        \[ 
            \Tr_X(\cl(Z) \cup f^*(\beta)) = \Tr_Y(\cl(f_*(Z)) \cup \beta)
        \]
        \item (Pullback) Let $f: X \to Y$ be flat of relative dimension $r$ and $Z \subseteq Y$ a closed irreducible subvariety with $[f^{-1}(Z)] = \sum_{i=1}^r m_iW_i$, then $f^*(\cl(Z)) = \sum_{i=1}^r m_i\cl(W_i)$. That is, $f^*(\cl(Z)) = \cl(f^*(Z))$. 
    \end{enumerate}
\end{definition}
\newpage
We define the pushforward map by using the Poincare duality: 
\begin{definition}(Pushforward)
    For $f: X \to Y$ a morphism and $\alpha \in H^i(X)$, define $f_*(\alpha) \in H^{i-(2m-2n)}(Y)$ to be: 
    \[ \Tr_Y(f_*(\alpha)\cup \beta) = \Tr_X(\alpha \cup f^*(\beta))\]
    for all $\beta \in H^{2m-i}(X)$. \\
    That is, using axiom 6, $f_*(\cl(Z)) = \cl(f_*(Z))$.
\end{definition}

\section{Grothendieck-Lefschetz Trace Formula:}

Given such a cohomology theory, we get the following theorems as a formal consequence: 
\begin{theorem}
    For $X$ non-singular, connected, projective variety:
    \begin{enumerate}
        \item $\cl: A^*(X)\to H^{2*}(X)$ is a ring map
        \item $f^*$ and $f_*$ are compute with this map. 
        \item For $\alpha_i \in A^{m_i}$ for $1 \leq i \leq r$ such that $\sum_i m_i = \dim(X)$, then 
        \[ (\alpha_1 \cdot \ldots \alpha_r) = \Tr(\cl(\alpha_1) \cup \cdots \cup \cl(\alpha_r))\]        
    \end{enumerate}
\end{theorem}

Additionally, using these axioms we can prove exactly what we want:
\begin{theorem}(Grothendieck-Lefschetz Trace Formula) If $\phi: X \to X$ is an endomorphism of the non-singular, connected, projective variety $X$, and if $\Gamma_\phi, \Delta \subseteq X \times X$ are the graph of $X \times X$ are the graph of $\phi$ and the diagonal, then:  
    \[ (\Gamma_\phi \cdot \Delta) = \sum_{i = 0}^{2n} (-1)^i \tr(\phi^*|H^i(X))\]
In particular, if $\Gamma_\phi$ and $\Delta$ intersect transversely, then the above expression computes $|\{x \in X | \phi(x) = x\}|$. 
\end{theorem}

\begin{proof}{Sketch of proof.}
    The proof follows analogously to Ribhu's. 
    \begin{enumerate}
        \item First use (3) above to translate the problem into cohomology.
        \item Use Kunneth to express cohomology classes in $H^*(X \times X)$ as pullbacks of those coming from $H(X)$.
        \item Various pullback and pushforward gymnastics to write things in terms of a pullbacks of a nice basis wrt Poincare duality.
        \item Use pushforward+product axiom and collect terms
    \end{enumerate}
\end{proof}

\section{Proof of Weil Conjecture I and II}


Given the Grothendieck-Lefschetz fixed point theorem, the proof of the rationality is reduced to linear algebra. 

\begin{theorem}
    $Z(t)$ is a rational function, i.e. $Z(t) \in \Q(t)$
\end{theorem}

\begin{proof}
    First, note that:
    \[ N_m = (\Gamma_{F^m} \cdot \Delta)\]
    where $\Gamma_{F^m}, \Delta \subseteq \overline{X} \times \overline{X}$.
    Via the GLFPT, we have:
    \[ N_m = \sum_{i = 0}^{2n} (-1)^i \tr(F^{*m}|H^i(\overline{X})) \]
    Thus, 
    \[ Z(X, t) = \exp\left(\sum_{m \geq 1} \sum_{i=0}^{2n} (-1)^i \tr(F^{*m} | H^i(\overline{X})) \frac{t^m}{m}\right)\]
\end{proof}

Then, we recall the following lemma from the very first talk. 
\begin{lemma}
    For every endomorphism $\phi$ of a finite dimensional vector space $V$ over a field $K$, we have:
    \[ \det(\Id - t\phi) = \exp\left(-\sum_{m\geq 1} \tr(\phi^m | V) \frac{t^m}{m}\right)\]
\end{lemma}
\begin{proof}
    Michaela showed parts of this in the very first talk. This is easy. \\
    Idea is to diagonalize $\phi$, rearrange, notice the Taylor expansion of $\log$ and conclude.  
\end{proof}

\begin{proof}{Proof of Theorem.}
    Continuing, we get: 
    \[  Z(X, t) = \prod_{i=0}^{2n} \det(\Id - tF^*| H^i(\overline{X}))^{(-1)^{i+1}}\]
    Thus, we conclude that $Z(X, t) \in K(t)$. We also know that $Z(X, t) \in \Q[[t]]$. \\
    This implies $Z(X, t) \in \Q(t)$. This is an easy exercise in linear algebra.
\end{proof}

We have the first Weil Conjecture (modulo Henry and Semir defining $l$-adic cohomology). \bbni
\newpage 

Now, we turn our attention to the functional equation. 
\begin{theorem}
    If $E$ is the Euler characteristic of $X$, then:
    \[ Z\left(X, \frac{1}{q^nt}\right) = \pm q^{nE/2}t^E Z(t) \]
\end{theorem}

To handle this, the key step is another linear algebra lemma: 
\begin{lemma}
    Let $\phi: V \times W \to K$ be a perfect pairing of vector spaces of dimension $r$ over the field $K$. If $\lambda \in K \setminus \{0\}$ and $f \in \End_K(V)$ and $g \in \End_K(W)$ are such that $\phi(f(v), g(w)) = \lambda \phi(v, w)$ for every $v \in V, w \in W$, then:
    \[ \det(\Id - tg|W) = \frac{(-\lambda t)^r}{\det(f | V)} \det(\Id - \lambda^{-1}t^{-1}f|V)\]
    and 
    \[ \det(g | W) = \frac{\lambda^r}{\det(f|V)}\]
\end{lemma}
\begin{proof}
    Mustata, lemma 4.15 or Hartshorne Appendix C, Lemma 4.3.  
\end{proof}

\begin{proof}[Proof of Theorem.]
    We want to apply the lemma to the Poincare pairing: 
    \[\phi_i : H^i(\overline{X}) \otimes H^{2n-i}(\overline{X}) \to H^{2n}(\overline{X}) \to K\] 
    defined by $\phi_i(\alpha, \beta) = \Tr(\alpha \cup \beta)$ and $f = F^*$ and $g = F^*$. Before that, we need to show there exists $\lambda \in K$ such that:
    \[ \phi_i(F^*(\alpha), F^*(\beta)) = \lambda \phi_i(\alpha, \beta) \]
    First notice that:
    \[ \phi_i(F^*(\alpha), F^*(\beta)) = \Tr_{\overline{X}}(F^*(\alpha \cup \beta))\]
    Thus, we need to show $F^*$ acts on $H^{2n}(\overline{X})$ by multiplication. \\
    First, we notice that $F$ is a finite morphism of degree $q^n$. \\
    Thus, for a point $Q \in \overline{X}$, $\deg([F^*Q]) = q^n$. We conclude by Axiom 7.
    \[ \phi_i(F^*(\alpha), F^*(\beta)) = q^n\Tr_{\overline{X}}(\alpha \cup \beta)) = q^n \phi_i(\alpha, \beta)\]
    Letting $P_i(t) = \det(\Id -tF^*| H^i(\overline{X}))$ and $B_i = \dim_K H^i(\overline{X})$, we use the lemma to conclude: 
    \[ \det(F^* | H^{2n-i}(\overline X)) = \frac{q^{nB_i}}{\det(F^*|H^i(\overline X))} \]
    and 
    \[ P_{2n-i}(t) = \frac{(-1)^{B_i}q^{nB_i}t^{B_i}}{\det(F^*|H^i(\overline{X}))} P_i(1/q^nt)\]
    This gives us: 
    \begin{align*}
        Z\left(X, \frac{1}{q^nt}\right) &= \prod_{i=0}^{2n} P_i\left(\frac{1}{q^nt}\right)^{(-1)^{i+1}} \\ 
        &= \prod_{i=0}^{2n} P_{2n-i}(t)^{(-1)^{i+1}} \cdot \frac{(-1)^Eq^{nE}t^E}{\prod_{i=0}^{2n} \det(F^*|H^i(\overline{X}))^{(-1)^i}} \\
        &= \pm Z(X, t) \cdot \frac{q^{nE}t^E}{q^{nE/2}} \\
        &= \pm q^{nE}t^E Z(X,t)
    \end{align*}
\end{proof}

We are done!
