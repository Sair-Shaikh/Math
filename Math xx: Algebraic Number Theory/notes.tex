\section{Ring of Integers}

\begin{definition}
    Let $L$ be a field containing an integral domain $A$. An element $\alpha$ of $L$ is said to be \textbf{integral} over $A$ if it is a root of a monic polynomial with coefficients in $A$.
\end{definition}
\bbni

\subsection{The Ring of Integers is a Ring:}

\begin{theorem}
    The elements of $L$ that are integral over $A$ form a ring.
\end{theorem}
\begin{proof}[Proof 1 (Sketch).]
    Let $f(X) = \sum a_ix^i \in A[X]$ be some polynomial with roots $\alpha_i$ in some larger field. 

    We know that $a_i$ are expressed in terms of elementary symmetric polynomials in the roots $\alpha_i$. Thus, $S_i(\alpha_1, \cdots, \alpha_n) \in A$. By the fundamental theorem of symmetric polynomials, all symmetric polynomials in $\alpha_i$ lie in $A$. 

    Then, we claim that if $\alpha_1, \cdots, \alpha_n$ in algebraically closed field $L$ are integral over $A$, then $g(\alpha_1, \cdots, \alpha_n)$ is integral over $A$ if $g$ has coefficients in $A$ and is monic. We do this by constructing, 
    \[ h(X) := \sum_{\sigma \in S_n} (X-g(\alpha_\sigma))\]
    This is monic. Its coefficients are symmetric polynomials in $\alpha_i$, hence in $A$. Moreover, $g(\alpha_1, \cdots, \alpha_n)$ is a root. 

    Now take $\alpha_1$ and $\alpha_2$ to be elements of $L$ integral over $A$. We can take a product of their defining polynomial to get a monic polynomial in $A[X]$ having both $\alpha_1$ and $\alpha_2$ as roots. Then, using the previous paragraph, we know that $\alpha_1 \pm \alpha_2$ and $\alpha_1\alpha_2$ are both integral. 
\end{proof}

\begin{theorem}
    Let $L$ be a field containing $A$. An element $\alpha \in L$ is integral over $A$ if and only if there exists a non-zero finitely generated $A$-module $M \subseteq L$ such that $\alpha M \subset M$. 
\end{theorem}

\begin{proof}
    The forward direction is easy because one can take the $A$-module generated by $\{1, \alpha, \cdots, \alpha^{n-1}\}$ where $n := \deg(f)$ and $f$ is monic with root $\alpha$. 
    
    For the reverse direction, you use Cramer's rule and some determinant shenanigans to get a monic polynomial with root $\alpha$ out. 
\end{proof}

\begin{proof}[Proof 2 of Thm 1.2]
    Let $\alpha$ and $\beta$ be two elements of $L$ integral over $A$. Let $M$ and $N$ be finitely generated $A$-submodules with $\alpha M \subset M$ and $\beta N \subset N$. Define, 
    \[ MN = \left\{ \sum m_in_i | m_i \in M, n_i \in N\right\}\]

    Then, $MN$ is an $A$-submodule of $L$. Moreover, it is finitely generated because the product of the generators of $M$ and $N$ generate $MN$. Moreover, it is stable under multiplication by $\alpha\pm \beta$ and $\alpha\beta$. Thus, by the previous theorem, $\alpha \pm \beta$ and $\alpha\beta$ are integral.
\end{proof}

\subsection{Basic Properties:}
\begin{definition}
    The ring of elements of $L$ integral over $A$ is called the \textbf{integral closure} of $A$ in $L$. The integral closure of $\Z$ in an algebraic number field $L$ is called the \textbf{ring of integers} $\OO_L$ in $L$. 
\end{definition}

\begin{proposition}
    Let $K$ be a field of fractions of $A$, and let $L$ be a field containing $K$. If $\alpha \in L$ is algebraic over $K$, then there exists a non-zero $d \in A$ such that $d\alpha$ is integral over $A$.
\end{proposition}
\begin{proof}
    If $\alpha$ is algebraic over $K$, it satisfies a monic polynomial equation $f(x) = x^n + \cdots + a_0$ with coefficients in $K$. Multiply through by $d^n$ where $d$ is the common denominator of the coefficients. You get, 
    \begin{align*}
        &d^n\alpha^n + a_{n-1}d^{n-1}\alpha^{n-1} + \cdots + d^ma_0 = 0 \\
        \implies &(d\alpha)^n + a_nd (d\alpha)^{n-1} + \cdots + d^ma_0 = 0 
    \end{align*}
\end{proof}

\begin{corollary}
    Let $A$ be an integral domain with field of fractions $K$, and let $B$ be the integral closure of $A$ in a field $L$ containing $K$. If $L$ is algebraic over $K$, then it is the field of fractions of $B$.  
\end{corollary}
\begin{proof}
    Apply the previous theorem to any $\alpha \in L$. You get $\alpha = \frac{\beta}{d}$ for $\beta \in B$ and $d \in A$. 
\end{proof}

\begin{definition}
    A ring $A$ is \textbf{integrally closed} if it is its own integral closure in its field of fractions $K$, i.e. $\alpha \in K$ integral over $A \implies \alpha \in A$. 
\end{definition}

\begin{proposition}
    A unique factorization domain is integrally closed. 
\end{proposition}
\begin{proof}
    Let $\frac{a}{b}$ with $a,b \in A$ be an element in the field of fractions of $A$ that is integral over $A$. If $b$ is a unit in $A$, we are done. Otherwise, there exists an irreducible element $\pi$ of $A$ dividing $b$ but not $a$. Then, take the polynomial, 
    \begin{align*}
        &\left(\frac{a}{b}\right)^n + c_{n-1}\left(\frac{a}{b}\right)^{n-1} +\cdots + c_0 = 0 \\ 
        \implies &a^n + c_{n-1}ba^{n-1} + \cdots + c_0b^n = 0
    \end{align*} 
    But then $\pi$ divides every term besides $a^n$, so it must divide $a^n$, thus it must divide $a$. This is a contradiction. 
\end{proof}

\begin{remark}
    We have found a criteria describing whether a ring is a UFD. That is, if it is not integrally closed, it cannot be a UFD.
\end{remark} 

\begin{example} \bbni
    \begin{enumerate}
        \item The rings $\Z$ and $\Z[i]$ are integrally closed because both are principal ideal domains.
        \item Unique factorization fails in $\Z[\sqrt{-3}]$ because $4 = 2 \times 2 = (1+\sqrt{-3})(1-\sqrt{-3})$ all irreducible. Similarly, the integral closure of $\Z[\sqrt{-3}]$ in $\Q[\sqrt{-3}]$ is $\Z[\zeta_3]$, which is a PID and bigger that $\Z[\sqrt{-3}]$. 
        \item For every field $k$, the integral closure of $k[S_1, \cdots, S_m]$ in $k(X_1, \cdots, X_m)$ is $k[X_1, \cdots, X_m]$. Being integral over $k[S_1, \cdots, S_m]$ implies integral over $k[X_1, \cdots, X_m]$ as $S_i$ are polynomials in $X_i$. Conversely, you can take an $f$ can similar to the proof of Theorem 1.2, create a monic symmetric polynomial in $X_i$ with $f(X_1, \cdots, X_n)$ as a root.
    \end{enumerate}
\end{example}

\begin{proposition}
    Let $K$ be the field of fractions of $A$, and let $L$ be an extension of $K$ of finite degree. Assume $A$ is integrally closed. An element $\alpha \in L$ is integral over $A$ if and only if its minimum polynomial over $K$ has coefficients in $A$. 
\end{proposition}
\begin{proof}
    The reverse direction is trivial For the forward, assume, 
    \[ \alpha^n + c_{n-1}\alpha^{n-1} + \cdots + c_0 = 0 \qquad c_i \in A\]
    Let $f(x)$ be the minimal polynomial for $\alpha$. Then, all the fields $K[\alpha']$ for $\alpha'$ root of $f(x)$ are ismorphic to $K[\alpha]$ via the automorphism $\sigma$ fixing $K$ and sending $\alpha' \mapsto \alpha$. Then, using $\sigma$ on the equation, we have that all conjugates of $\alpha$ are also integral. 

    Then, the coefficients $c_i$ are polynomials in the roots of $f(x)$. Thus, they are all integral over $A$ as the ring of integers is a ring. Moreover, as they lie in $K$, and $A$ is integrally closed, integral over $A$ implies being in $A$. 
\end{proof}

\begin{proposition}
    If $B$ is integral over $A$ and finitely generated as an $A$-algebra, then it is finitely generated as an $A$-module. 
\end{proposition}
\begin{proof}
    First assume $B = A[\beta]$ for some $\beta$ integral over $A$. Then $\beta$ satisfies a polynomial $f$ with coefficients in $A$. Let $n := \deg(f)$. Then $\{1, \cdots, \beta^{n-1}\}$ generate $B$ as an $A$-module. We complete by induction, noting that $A[\beta_1,\beta_2] = A[\beta_1][\beta_2]$ and that $\beta_2$ is integral $A[\beta_1]$ as its integral over $A$. 
\end{proof}

\begin{proposition}
    Consider integral domains $A \subset B \subset C$. If $B$ is integral over $A$ and $C$ is integral over $B$, then $C$ is integral over $A$. 
\end{proposition}
\begin{proof}
    Let $\gamma \in C$ be an element integral over $B$. Then it satisfies, 
    \[ \gamma^n + b_{n-1}\gamma^{n-1} + \cdots + b_0 = 0\]
    Thus, $\gamma$ is integral over $B' = A[b_0, \cdots, b_{n-1}]$. Moreover, $B'[\gamma]$ is integral over $A$ and finitely-generated as an $A$-algebra, thus finitely generated as a $A$-module. Moreover, $\gamma B'[\gamma] \subset B'[\gamma]$ via the equation above. 
\end{proof}

\begin{corollary}
    The integral closure of $A$ in an algebraic extension $L$ of its field of fractions $K$ is integrally closed. 
\end{corollary}
\begin{proof}
    Let $B$ be the integral closure of $A$ over $L$. Let $C$ be the integral closure of $B$. Then, $C \subset B$ by the previous proposition.  
\end{proof}

\begin{remark}
    The ring of integers in a number field is integrally closed. We want this because we want our ring of integers to have the best chance of being a UFD. 
\end{remark}

\begin{example}
    Let $k$ be a finite field, and let $K$ be a finite extension of $k(X)$. Let $\OO_K$ be the integral closure of $k[X]$ in $K$. The arithmetic of $\OO_K$ is very similar to that of the ring of integers in a number field.
\end{example}

\begin{proposition}
    Let $A$ be an integrally closed integeral domain and let $L$ be a finite extension of the field of fractions $K$ of $A$. If $\beta \in L$ is integral over $A$, then $\Tr_{L/K}(\beta)$ and $\Nm_{L/K}(\beta)$ are in $A$.
\end{proposition} 

\begin{proposition}
    Let $K$ be the field of fractions of $A$ and let $L$ be a finite seperable extension of $K$ of degree $m$. If the integral closure $B$ of $A$ in $L$ is free of rank $m$ over $A$, then $\disc(B/A) \neq 0$. 
\end{proposition}


\subsection{Rings of Integers are finitely-generated}


\begin{proposition}
    Let $A$ be an integrally closed integral domain with field of fractions $K$, and let $B$ be the integral closure of $A$ is a separable extension $L$ of $K$ of degree $m$. There exist free $A$-submodules $M$ and $M'$ of $L$ such that, 
    \[ M \subset B \subset M' \]
    Therefore $B$ is a finitely generated $A$-module if $A$ is Noetherian, and it is free of rank $m$ if $A$ is a PID. 
\end{proposition}
\begin{proof}
    Pick $\{\beta_i\}$ to be a basis for $L$ over $K$. We know that $d_i\beta_i$ is integral over $A$, i.e. is in $B$ for some $d_i \in A$. Given $B$ is a ring, we can even pick the same $d$. So, $\{d\beta_i\}$ is also a basis. Thus, without loss of generality, we let $\beta_i \in B$. 

    We use the non-degenerate trace pairing to get a dual basis $\{\beta_i'\}$ of $L$ over $K$ with $\Tr(\beta_i \cdot \beta_j') = \delta_{ij}$. Now, we claim,
    \[ A\beta_1 + \cdots A\beta_m \subset B \subset A\beta_1' + \cdots A\beta_m' \]

    The first containment is trivial as $\beta_i \in B$ and $A \subset B$. For the second containment, let $\beta \in B$. Then, we can write, $\beta = \sum b_j\beta_j'$ with $b_j \in K$. We have to show that $b_j \in A$.
    
    Since $\beta_i, \beta \in B$, so is $\beta\beta_i$. Thus, $\Tr(\beta\beta_i) \in A$. But, 
    \begin{align*}
        \Tr(\beta \cdot \beta_i) = \Tr\left(\sum_j b_j\beta_j', \beta\right) = \sum_j b_j \delta_{ij} = b_i
    \end{align*}
    Hence, $b_i \in A$. 

    If $A$ is Noetherian, $M'$ is a Noetherian $A$-module and thus $B$ is finitely generated as an $A$-module. If $A$ is a PID, then $B$ is free of rank $= m$ because it is contained in and contains a free rank $m$ module.
\end{proof}


\begin{corollary}
    The ring of integers in a number field $L$ is the largest subring that is finitely generated as a $\Z$-module. 
\end{corollary}
\begin{proof}
    By the previous theorem, $\OO_L$ is a finitely generated $\Z$-module. If $B$ is another subring of $L$ that is a finitely generated $\Z$-module, then every element $\alpha \in B$ satisfies $\alpha \Z[\alpha] \subset \Z$. Thus, $\alpha$ is integral and thus in $\OO_L$.  
\end{proof}


\begin{definition}
    When $K$ is a number field, a basis $\alpha_1, \cdots, \alpha_m$ for $\OO_K$ as a $\Z$-module is called an \textbf{integral basis} for $K$.    
\end{definition}

\begin{exercise}{2-1}
Since $\mathbb{Z}[\sqrt{5}]$ is not integrally closed, it can not be a unique factorization domain. Give an example of an element of $\mathbb{Z}[\sqrt{5}]$ that has two distinct factorizations into irreducible elements.
\end{exercise}

\begin{exercise}{2-2}
Let $A$ be an integrally closed ring, and let $K$ be its field of fractions. Let $f(X)\in A[X]$ be a monic polynomial. If $f(X)$ is reducible in $K[X]$, show that it is reducible in $A[X]$.
\end{exercise}

\begin{exercise}{2-3}
Show that if $L/K$ is not separable, then $\mathrm{disc}(L/K)=0$.
\end{exercise}

\begin{exercise}{2-4}
Let $\mathfrak{a}=(2,1+\sqrt{-3})$ in $\mathbb{Z}[\sqrt{-3}]$. Show that $\mathfrak{a}\neq (2)$, but $\mathfrak{a}^2=(2)\mathfrak{a}$. Conclude that ideals in $\mathbb{Z}[\sqrt{-3}]$ do not factor uniquely into prime ideals. (Hence $\mathbb{Z}[\sqrt{-3}]$ is the wrong choice for the ring of integers in $\mathbb{Q}[\sqrt{-3}]$.)
\end{exercise}

\end{exercise}

\begin{exercise}{2-6}
Let $K=\mathbb{Q}[\sqrt{7},\sqrt{10}]$, and let $\alpha$ be an algebraic integer in $K$. The following argument will show that $\mathcal{O}_K\neq \mathbb{Z}[\alpha]$.

(a) Consider the four algebraic integers:
\begin{align*}
    \alpha_1=(1+\sqrt{7})(1+\sqrt{10}),\quad
    \alpha_2=(1+\sqrt{7})(1-\sqrt{10}), \\
    \alpha_3=(1-\sqrt{7})(1+\sqrt{10}),\quad
    \alpha_4=(1-\sqrt{7})(1-\sqrt{10})    
\end{align*}
Show that all the products $\alpha_i\alpha_j$, $i\neq j$, are divisible by $3$ in $\mathcal{O}_K$, but that $3$ does not divide any power of any $\alpha_i$. [Hint: Show that $\alpha_i^n/3$ is not an algebraic integer by considering its trace: show that $\mathrm{Tr}(\alpha_i^n)=\left(\sum \alpha_j^n\right)\equiv 4^n\pmod{3}$ in $\mathbb{Z}[\alpha]$; deduce $\mathrm{Tr}(\alpha_i^n)\equiv 1\pmod{3}$ in $\mathbb{Z}$.]

(b) Assume now that $\mathcal{O}_K=\mathbb{Z}[\alpha]$ — we shall derive a contradiction. Let $f(X)$ be the minimum polynomial of $\alpha$ over $\mathbb{Q}$. For $g(X)\in \mathbb{Z}[X]$, let $\bar{g}(X)$ denote the image of $g$ in $\mathbb{F}_3[X]$, $\mathbb{F}_3=\mathbb{Z}/(3)$. Show that $g(\alpha)$ is divisible by $3$ in $\mathbb{Z}[\alpha]$ if and only if $\bar{g}$ is divisible by $\bar{f}$ in $\mathbb{F}_3[X]$.

(c) For each $i$, $1\leq i\leq 4$, let $f_i$ be a polynomial in $\mathbb{Z}[X]$ such that $\alpha_i=f_i(\alpha)$. Show that $\bar{f}\mid \bar{f_i}\bar{f_j}$ $(i\neq j)$ in $\mathbb{F}_3[X]$, but that $\bar{f}$ does not divide $\bar{f_i}^n$ for any $n$. Conclude that for each $i$, $\bar{f}$ has an irreducible factor which does not divide $\bar{f_i}$ but does divide all $\bar{f_j}$, $j\neq i$.

(d) This shows that $\bar{f}$ has at least four distinct irreducible factors over $\mathbb{F}_3$. On the other hand, $f$ has degree at most $4$. Why is this a contradiction?
\end{exercise}

\begin{exercise}{2-7}
Let $A$ be an integral domain, and let $B$ be the integral closure of $A$ in a finite extension $L$ of its field of fractions $K$. Let $S$ be a multiplicative subset of $A$. Show that $S^{-1}B$ is the integral closure of $S^{-1}A$ in $L$.
\end{exercise}

Let $\mathfrak{p}$ be a prime ideal in an integral domain $A$. Show that $A_{\mathfrak{p}}/\mathfrak{p}A_{\mathfrak{p}}$ is the field of fractions of $A/\mathfrak{p}A$.
\end{exercise}
