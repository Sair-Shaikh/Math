\section{Line Bundles and Divisors:}

Recall, the following bijection, 
\begin{align*}
    \Mor(X, \P^r) \leftrightarrow \{(\LL, s_0, \cdots, s_r) : \LL \text{ line bundle}, s_i \in \Gamma(\LL), \text{ basepoint free}\}
\end{align*}

Two of the central problems in Algebraic Geometry are:
\begin{enumerate}
    \item Classify sheaves on a scheme $X$, in particular, line bundles.
    \item Understand $\Gamma: \Coh(X) \to \Ab$. 
\end{enumerate}

\begin{definition}
    Let $\Pic(X) = \{\text{line bundles}\}/\text{isom.}$, equipped with the group operation $\otimes$. In fact, $\Pic$ is a contravariant functor as, 
    \begin{align*}
        f: X \to Y \implies f^*: \Pic(Y) \to \Pic(X)
    \end{align*}
    with $f^*([\LL]) = [f^*\LL]$. 
\end{definition}

In topology, we know how to classify line bundles. In fact, 
\[ \Pic(X) = [X, \CP^\infty] = [X, K(\Z, 2)] = H^2_{sing}(X)\]

Here are some trivial examples. 
\begin{enumerate}
    \item $\Pic(\Spec k) = 1$
    \item $\Pic(\A^1_k) = 1$
    \item More generally, $\Pic(\Spec A) = 1$ for $A$ a PID, since rank $1$ projective modules over $A$ are free.  
\end{enumerate}

Moving towards less trivial examples, we ask the question, where does a section of a line bundle $\LL$ vanish. 

Locally, a section of $\LL$ is a function. Thus, it should vanish along a divisor (i.e. $\codim 1$).

For simplicity, assume $X$ is Noetherian, normal, and irreducible. 

\begin{definition}
    A \textbf{Weil Divisor} is a formal $\Z$-linear combination, 
    \[ \sum_{\substack{ Y \subset X  \\ \codim Y = 1 \\ irred+red}} n_Y [Y]\]
    where all but finitely many $n_Y$ are zero. 

    Let $\Weil(X)$ be the abelian group of Weil divisors. 
\end{definition}

\begin{remark}
    In topology, these play the role of $2$-cocycles (via Poincare duality)
\end{remark}

\begin{definition}
    \begin{enumerate}
        \item A divisor of the form $D = [Y]$ is called \textbf{irreducible}.
        \item A divisor $D = \sum n_Y [Y]$ is \textbf{effective} if $n_Y \geq 0$ for all $Y$. We notate this as $D > 0$. 
        \item We write $D_1 \geq D_2$ if $D_1 - D_2 \geq 0$. This is a partial order on $\Weil(X)$. 
    \end{enumerate}
\end{definition}

\begin{example}
    \begin{enumerate}
        \item If $X$ is a curve, $\Weil(X)$ is a free abelian group on the closed points of $X$. 
        \item $X = \P^r$. Then irreducible divisors are vanishing of single irreducible homogeneous polynomials. 
    \end{enumerate}
\end{example}

\begin{definition}
    Let $\LL$ be a line bundle on $X$ and $s$ a rational section not vanishing identically on any component. Then, we define the \textbf{divisor of zeros and poles}. 
    \[ \div s = \sum_Y \val_Y(s)[Y]\]
    Since a section of $\LL$ is locally a function (under any trivializion in any open neighborhood of $Y$.)
\end{definition}

Note that $\{(\LL, s)\}$ under $\otimes$ with $(\LL, s) \otimes (\LL', s') = (\LL \otimes \LL', s\otimes s')$ is an Abelian group. Furthermore, we have that
\[ \div: \{ (\LL, s)\}/\text{isom} \to \Weil(X)\]
is a homomorphism.

\begin{example}
    Let $\LL = \OO_{\P^n}(m)$. Let $f = \frac{g_1-g_2^3}{g_3^2}$ with $\deg g_1 + 3\deg g_2 - 2\deg g_3 = m$. Then, 
    \[ \div(\OO(m), f) = [\V(g_1)] + 3[\V(g_2)] - 2[\V(g_3)]\]
\end{example}

\begin{example}
    Let $\LL$ be the ideal sheaf of the origin in $\A^1$. Then, 
    \[ \div\left(\LL, \frac{x^3}{x-1}\right) = 2[0] - [1]\]
    Why? The key point is that $\LL$ and $\OO$ in a neighborhood of $x = 0$ is division by $x$. That is $\frac{x^2}{x-1}$ vanishes at $x = 0$ vanishes to order $2$ more than it was asked to. 
\end{example}

\begin{proposition}
    If $X$ is noetherian and normal, then $\div$ is injective. 
\end{proposition}

\begin{proof}
    Why? $\div(\LL, s) = 0$ means that $s$ has no zeros and no poles as a section of $\LL$. That is, $s$ and $\frac{1}{s}$ are both regular. Thus, the map $\cdot s:\OO_X \to\LL$ is an isomorphism. That is $(\OO_X, 1) \cong (\LL, s)$. 
\end{proof}

Next, we will see that $\div$ is surjective. 

\begin{definition}
    Let $D$ be a Weil divisor on $X$. Define sheaf $\OO_X(D)$ via the following: 
    \[ \Gamma(U, \OO_X(D)) = \{t \in K(X)^\times.: \div|_U t + D|_U \geq 0 \} \cup \{0\}\]
\end{definition}


\begin{example}
    If $D = -[\text{origin}]$ in $\A^1$. Then $\OO_X(D)$ is the ideal sheaf of the origin. For $U$ not containing the origin, having $\div|_U t \geq 0$ means that $t$ is regular. If $U$ contains the origin, then $\div|_U t -[\text{origin}]|_U \geq 0$ means it is regular and vanishes at the origin
\end{example}
